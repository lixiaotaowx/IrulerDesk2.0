cmake_minimum_required(VERSION 3.16)
project(WebSocketServerWithRouting)

# 设置C++标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 查找Qt6
find_package(Qt6 REQUIRED COMPONENTS Core WebSockets)

# 启用Qt的MOC
set(CMAKE_AUTOMOC ON)

# 创建可执行文件
add_executable(websocket_server_routing websocket_server_with_routing.cpp)

# 链接Qt库
target_link_libraries(websocket_server_routing Qt6::Core Qt6::WebSockets)

# 设置输出目录
set_target_properties(websocket_server_routing PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)

# 如果是Release模式，添加优化选项
if(CMAKE_BUILD_TYPE STREQUAL "Release")
    if(MSVC)
        target_compile_options(websocket_server_routing PRIVATE /O2)
    else()
        target_compile_options(websocket_server_routing PRIVATE -O3)
    endif()
endif()

# 安装规则
install(TARGETS websocket_server_routing
    RUNTIME DESTINATION bin
)
